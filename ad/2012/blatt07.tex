\subsubsection*{2.}
\begin{verbatim}
Münzen-Ausgeben(x W R)
    sort(W)
    current_w = W.pop
    
    while x-current_w >= 0
        x = x - current_w
        R.append(current_w)
    
    if x > 0
        Münzen-Ausgeben(x W R)
    
    return R                                                                                           
\end{verbatim}
